## Status Detection Logic

```typescript
function getResponseStatus(
  turn: ConversationTurn,
  dialogStatus: DialogStatusType,
  isLastTurn: boolean
): ResponseStatus {
  // No response object
  if (!turn.response) {
    if (isLastTurn && dialogStatus === DialogStatus.IN_PROGRESS) {
      return ResponseStatus.IN_PROGRESS;
    }
    return ResponseStatus.PENDING;
  }

  // Check for error indicators
  if (turn.response.message === "" && hasErrorInChatData) {
    return ResponseStatus.ERROR;
  }

  // Response exists with content
  if (turn.response.message) {
    return ResponseStatus.SUCCESS;
  }

  return ResponseStatus.PENDING;
}
```
